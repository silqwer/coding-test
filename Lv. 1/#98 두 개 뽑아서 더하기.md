int solution(char* t, char* p) {

    int count = 0;
    int len_t = strlen(t);
    int len_p = strlen(p);
    long long num_p = atoll(p);
    
    for (int i = 0; i <= len_t - len_p; i++) {
        char sub_str[len_p + 1];
        strncpy(sub_str, t + i, len_p);
        sub_str[len_p] = '\0';
        long long num_sub_str = atoll(sub_str);
        if (num_sub_str <= num_p) {
            count++;
        }
    }
    
    return count;
}
