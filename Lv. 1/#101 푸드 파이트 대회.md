char* solution(int food[], size_t food_len){

    int sum = 0;
    for (size_t i = 1; i < food_len; i++) {
        sum += food[i];
    }
    int half_sum = sum / 2;

    char* answer = (char*) malloc(sum + 1);
    for (int i = 0; i < sum; i++) {
        if (food[1] > 0) {
            answer[i] = '1';
            food[1]--;
        } else if (food[2] > 0) {
            answer[i] = '2';
            food[2]--;
        } else {
            answer[i] = '3';
            food[3]--;
        }
        if (i >= half_sum) {
            if (food[1] > 0) {
                answer[i] = '1';
                food[1]--;
            } else if (food[2] > 0) {
                answer[i] = '2';
                food[2]--;
            } else {
                answer[i] = '3';
                food[3]--;
            }
        }
    }
    answer[sum] = '\0';

    return answer;
}
