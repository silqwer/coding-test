bool compare(string a, string b, int n) {
    if (a.at(n) != b.at(n)) {
        return a.at(n) < b.at(n);
    }
    else {
        return a < b;
    }
}

vector<string> solution(vector<string> strings, int n) {
    sort(strings.begin(), strings.end(), [n](string a, string b) { return compare(a, b, n); });
    return strings;
}
