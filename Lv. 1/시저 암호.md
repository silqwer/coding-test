#include <string>
#include <vector>

using namespace std;

string solution(string s, int n) {
    string answer = "";
    for(char c : s) {
        if(c == ' ') {
            answer += c;
        }
        else if(isupper(c)) {
            answer += ((c - 'A' + n) % 26 + 'A');
        }
        else {
            answer += ((c - 'a' + n) % 26 + 'a');
        }
    }
    return answer;
}
