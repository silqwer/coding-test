func solution(arr1 [][]int, arr2 [][]int) [][]int {
    rows, cols := len(arr1), len(arr1[0])
    result := make([][]int, rows)
    for i := 0; i < rows; i++ {
        result[i] = make([]int, cols)
        for j := 0; j < cols; j++ {
            result[i][j] = arr1[i][j] + arr2[i][j]
        }
    }
    return result
}
