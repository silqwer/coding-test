vector<string> solution(int n, vector<int> arr1, vector<int> arr2) {
    vector<string> answer;

    for (int i = 0; i < n; i++) {
        string str = "";
        int num = arr1[i] | arr2[i];
        for (int j = 0; j < n; j++) {
            if (num % 2 == 1) {
                str = "#" + str;
            } else {
                str = " " + str;
            }
            num /= 2;
        }
        answer.push_back(str);
    }

    return answer;
}
